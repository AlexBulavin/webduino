<script>
	var myToolbox = "https://fustyles.github.io/webduino/LinkIt7697/test/toolbox.xml";
	var myEN = "https://fustyles.github.io/webduino/LinkIt7697/test/en.js";
    	var myCategoryID = "TEST";

	var xml = $.ajax({url: myToolbox, async: false}).responseXML;
	var toolbox = document.getElementById('toolbox');
	toolbox.children["category_external"].append(xml.firstChild);
	var myCategory = document.getElementById(myCategoryID);
	myCategory.setAttribute("name",Blockly.Msg.TEST_SHOW);  // Category Name = Blockly.Msg.xxxxx

	var myTimer = setTimeout(function(){
		if (document.getElementById('select-lang-en').checked!=document.getElementById('select-lang-zh').checked) {
			if ((document.getElementById('select-lang-en').checked)) {
				var s = document.createElement("script");
				s.type = "text/javascript";
				s.src = myEN;
				$("body").append(s);
				setTimeout(function(){
					myCategory.setAttribute("name",Blockly.Msg.TEST_SHOW);  // Category Name = Blockly.Msg.xxxxx
					Blockly.updateToolbox(toolbox);
				}, 2000);
			}
			clearInterval(myTimer);
		}
	}, 100); 
</script>
