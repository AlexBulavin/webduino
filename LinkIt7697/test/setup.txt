<!-- 
https://www.facebook.com/francefu    2020/2/23 10:30
適用 LinkIt 7697 BlocklyDuino v3

GitHub上放置五個檔案(blocks.js, javascript.js, toolbox.xml, en.js, zh-hant.js)，
程式碼中有中文註解的部分要修改成自訂積木的設定。
若是要新增數個自訂積木，有中文註解的程式原處複製修改成數個自訂積木設定多次執行。
將程式碼置於BlocklyDuino3\package.nw\main.html中</html>之後。
-->
<script>
	var toolbox = document.getElementById('toolbox');
	
	var zhhant_path = "https://fustyles.github.io/webduino/LinkIt7697/test/zh-hant.js";  //載入自訂積木繁體語系設定檔(預設繁體語系)
	var blocks_path = "https://fustyles.github.io/webduino/LinkIt7697/test/blocks.js";   //載入自訂積木定義檔
	var javascript_path = "https://fustyles.github.io/webduino/LinkIt7697/test/javascript.js";   //載入自訂積木轉出程式碼檔
	var toolbox_path = "https://fustyles.github.io/webduino/LinkIt7697/test/toolbox.xml";  //讀取自訂積木目錄檔
	var fatherCategoryID = "category_external";    //設定自訂積木的目錄(toolbox.xml)要掛載的官方主目錄Category ID為"category_external" (main.html)
	initial(zhhant_path, blocks_path, javascript_path, toolbox_path, fatherCategoryID);  //載入自訂積木

	var myTimer = setInterval(function(){
		if (document.getElementById('select-lang-en').checked!=document.getElementById('select-lang-zh').checked) {
			clearInterval(myTimer);
			if ((document.getElementById('select-lang-en').checked)) {
				addScript("https://fustyles.github.io/webduino/LinkIt7697/test/en.js");  //載入自訂積木英文語系設定檔覆蓋繁體語系設定
			}
			setTimeout(function(){
				setCategoryName("TEST", Blockly.Msg.TEST_SHOW);  //設定自訂積木Category ID為"TEST"的目錄(toolbox.xml)顯示名稱 = Blockly.Msg.TEST_SHOW (語系.js)
				Blockly.updateToolbox(toolbox);  //目前會造成目錄大風吹，可考慮不執行這行程式碼
			}, 1000);
		}
	}, 100); 

	function addScript(url) {
		var s = document.createElement("script");
		s.type = "text/javascript";
		s.src = url;
		$("body").append(s);
	}

	function setCategoryName(categoryID, msgName) {
		document.getElementById(categoryID).setAttribute("name", msgName);
	}

	function initial(zhhant_path, blocks_path, javascript_path, toolbox_path, fatherCategoryID) {
		addScript(zhhant_path);
		addScript(blocks_path);
		addScript(javascript_path);
		var xml = $.ajax({url: toolbox_path, async: false}).responseXML;
		toolbox.children[fatherCategoryID].append(xml.firstChild);
	}
</script>
